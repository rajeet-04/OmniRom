from fastapi import APIRouter, HTTPException
from pydantic import BaseModel, Field
from typing import List
import asyncio
from concurrent.futures import ThreadPoolExecutor
import time

from app.engines.router import route_romanization
from app.core.detector import detect_script, get_script_type
from app.core.cache import get_cache

router = APIRouter(prefix="/v1/romanize", tags=["batch"])

class BatchRequest(BaseModel):
    texts: List[str] = Field(..., description="List of texts to romanize", max_length=100)
    style: str = Field("standard", description="Romanization style")

class BatchItemResult(BaseModel):
    original: str
    romanized: str
    metadata: dict

class BatchResponse(BaseModel):
    results: List[BatchItemResult]
    total: int
    cached_count: int
    processing_time_ms: int


def romanize_single(text: str, style: str, cache) -> BatchItemResult:
    """Process a single text."""
    detection = detect_script(text)
    script_type = get_script_type(detection["language"], detection["script"])
    script_code = detection.get("script_code", "")
    
    # Check cache
    cached_result = cache.get(text, style)
    is_cached = False
    
    if cached_result:
        romanized = cached_result["romanized"]
        is_cached = True
    else:
        romanized, engine_used = route_romanization(text, script_type, script_code, style)
        # Cache it
        metadata = {
            "detected_lang": detection["language"],
            "engine_used": engine_used,
        }
        cache.set(text, romanized, metadata, style)
    
    return BatchItemResult(
        original=text,
        romanized=romanized,
        metadata={
            "detected_lang": detection["language"],
            "cached": is_cached,
        }
    )


@router.post("/batch", response_model=BatchResponse)
async def romanize_batch(request: BatchRequest):
    """Romanize multiple texts in a single request."""
    if not request.texts:
        raise HTTPException(status_code=400, detail="Texts list cannot be empty")
    
    if len(request.texts) > 100:
        raise HTTPException(status_code=400, detail="Maximum 100 texts per batch")
    
    start_time = time.time()
    cache = get_cache()
    
    # Process in parallel using thread pool
    results = []
    with ThreadPoolExecutor(max_workers=10) as executor:
        futures = [
            executor.submit(romanize_single, text, request.style, cache)
            for text in request.texts
        ]
        results = [f.result() for f in futures]
    
    cached_count = sum(1 for r in results if r.metadata.get("cached", False))
    processing_time_ms = int((time.time() - start_time) * 1000)
    
    return BatchResponse(
        results=results,
        total=len(results),
        cached_count=cached_count,
        processing_time_ms=processing_time_ms
    )

from fastapi import APIRouter, HTTPException, UploadFile, File, Form
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from typing import Optional
import io
import csv
import time

from app.engines.router import route_romanization
from app.core.detector import detect_script, get_script_type

router = APIRouter(prefix="/v1/romanize", tags=["files"])

ALLOWED_EXTENSIONS = {'.txt', '.csv', '.srt'}
MAX_FILE_SIZE = 10 * 1024 * 1024  # 10MB


@router.post("/file")
async def romanize_file(
    file: UploadFile = File(...),
    style: str = Form("standard")
):
    """Process a text file and return romanized content."""
    
    # Validate file extension
    import os
    ext = os.path.splitext(file.filename)[1].lower()
    if ext not in ALLOWED_EXTENSIONS:
        raise HTTPException(
            status_code=400,
            detail=f"File type not supported. Allowed: {ALLOWED_EXTENSIONS}"
        )
    
    # Read file content
    content = await file.read()
    if len(content) > MAX_FILE_SIZE:
        raise HTTPException(status_code=400, detail="File too large (max 10MB)")
    
    try:
        text = content.decode('utf-8')
    except UnicodeDecodeError:
        raise HTTPException(status_code=400, detail="Invalid UTF-8 encoding")
    
    # Process based on file type
    if ext == '.txt':
        result = await _process_text_file(text, style)
    elif ext == '.csv':
        result = await _process_csv_file(text, style)
    elif ext == '.srt':
        result = await _process_srt_file(text, style)
    
    # Return as downloadable file
    output = io.BytesIO(result.encode('utf-8'))
    return StreamingResponse(
        output,
        media_type="text/plain",
        headers={
            "Content-Disposition": f"attachment; filename=romanized_{file.filename}"
        }
    )


async def _process_text_file(text: str, style: str) -> str:
    """Process plain text file line by line."""
    lines = text.split('\n')
    results = []
    
    for line in lines:
        if line.strip():
            detection = detect_script(line)
            script_type = get_script_type(detection["language"], detection["script"])
            script_code = detection.get("script_code", "")
            romanized, _ = route_romanization(line, script_type, script_code, style)
            results.append(romanized)
        else:
            results.append(line)
    
    return '\n'.join(results)


async def _process_csv_file(text: str, style: str) -> str:
    """Process CSV file - romanize all string values."""
    import csv
    import io
    
    reader = csv.reader(io.StringIO(text))
    output = io.StringIO()
    writer = csv.writer(output)
    
    for row in reader:
        new_row = []
        for cell in row:
            if cell.strip() and _looks_like_text(cell):
                detection = detect_script(cell)
                script_type = get_script_type(detection["language"], detection["script"])
                script_code = detection.get("script_code", "")
                romanized, _ = route_romanization(cell, script_type, script_code, style)
                new_row.append(romanized)
            else:
                new_row.append(cell)
        writer.writerow(new_row)
    
    return output.getvalue()


async def _process_srt_file(text: str, style: str) -> str:
    """Process SRT subtitle file."""
    import re
    
    # SRT format: number -> timestamp -> text -> blank
    result_lines = []
    blocks = text.strip().split('\n\n')
    
    for block in blocks:
        lines = block.split('\n')
        if len(lines) >= 3:
            # Keep number and timestamp
            result_lines.append(lines[0])  # Number
            result_lines.append(lines[1])  # Timestamp
            
            # Romanize subtitle text
            subtitle_text = ' '.join(lines[2:])
            detection = detect_script(subtitle_text)
            script_type = get_script_type(detection["language"], detection["script"])
            script_code = detection.get("script_code", "")
            romanized, _ = route_romanization(subtitle_text, script_type, script_code, style)
            result_lines.append(romanized)
            result_lines.append('')  # Blank line
        else:
            result_lines.append(block)
    
    return '\n'.join(result_lines)


def _looks_like_text(text: str) -> bool:
    """Heuristic: looks like text needing romanization."""
    # Check if contains non-ASCII letters
    return any(ord(c) > 127 for c in text)
