```python
import pytest
from app.core.detector import detect_script, get_script_type

class TestDetector:
    """Tests for script detection."""
    
    def test_detect_cyrillic(self):
        """Test Cyrillic detection."""
        result = detect_script("Привет мир")
        assert result["script"].lower() == "cyrillic"
        assert result["reliable"] is True
    
    def test_detect_greek(self):
        """Test Greek detection."""
        result = detect_script("Γειά σου")
        assert result["script"].lower() == "greek"
    
    def test_detect_arabic(self):
        """Test Arabic detection."""
        result = detect_script("مرحبا")
        assert result["script"].lower() == "arabic"
    
    def test_detect_hindi(self):
        """Test Hindi/Devanagari detection."""
        result = detect_script("नमस्ते")
        assert result["script"].lower() == "deva"
    
    def test_detect_chinese(self):
        """Test Chinese detection."""
        result = detect_script("你好世界")
        assert result["script"] in ["hans", "hant", "hani"]
    
    def test_detect_japanese(self):
        """Test Japanese detection."""
        result = detect_script("東京")
        assert result["script"].lower() in ["jpan", "hani"]
    
    def test_detect_korean(self):
        """Test Korean detection."""
        result = detect_script("한글")
        assert result["script"].lower() == "hangul"
    
    def test_detect_latin(self):
        """Test Latin script passes through."""
        result = detect_script("Hello World")
        # May detect as English
        assert result["language"] is not None
    
    def test_empty_text(self):
        """Test empty text handling."""
        result = detect_script("")
        assert result["language"] == "und"
        assert result["reliable"] is False
    
    def test_get_script_type_cyrillic(self):
        """Test script type mapping for Cyrillic."""
        assert get_script_type("ru", "Cyrillic") == "cyrillic"
    
    def test_get_script_type_greek(self):
        """Test script type mapping for Greek."""
        assert get_script_type("el", "Greek") == "greek"
    
    def test_get_script_type_unknown(self):
        """Test script type for unknown scripts."""
        assert get_script_type("xx", "Unknown") == "other"
```

```python
import pytest
from fastapi.testclient import TestClient
from app.main import app

client = TestClient(app)


class TestHealthEndpoint:
    """Tests for health endpoints."""
    
    def test_root(self):
        """Test root endpoint."""
        response = client.get("/")
        assert response.status_code == 200
        assert "OmniRom" in response.json()["message"]
    
    def test_health(self):
        """Test health endpoint."""
        response = client.get("/health")
        assert response.status_code == 200
        assert response.json()["status"] == "healthy"


class TestRomanizeEndpoint:
    """Tests for romanization endpoint."""
    
    def test_romanize_cyrillic(self):
        """Test Cyrillic romanization."""
        response = client.post("/v1/romanize", json={
            "text": "Привет"
        })
        assert response.status_code == 200
        data = response.json()
        assert "romanized" in data
        assert "metadata" in data
        assert data["metadata"]["detected_lang"] == "ru"
    
    def test_romanize_hindi(self):
        """Test Hindi romanization."""
        response = client.post("/v1/romanize", json={
            "text": "नमस्ते"
        })
        assert response.status_code == 200
    
    def test_romanize_chinese(self):
        """Test Chinese romanization."""
        response = client.post("/v1/romanize", json={
            "text": "你好"
        })
        assert response.status_code == 200
    
    def test_romanize_japanese(self):
        """Test Japanese romanization."""
        response = client.post("/v1/romanize", json={
            "text": "東京"
        })
        assert response.status_code == 200
    
    def test_romanize_empty(self):
        """Test empty text returns 400."""
        response = client.post("/v1/romanize", json={
            "text": ""
        })
        assert response.status_code == 400
    
    def test_romanize_batch(self):
        """Test batch processing."""
        response = client.post("/v1/romanize/batch", json={
            "texts": ["Привет", "Hello", "东京"]
        })
        assert response.status_code == 200
        data = response.json()
        assert len(data["results"]) == 3
        assert data["total"] == 3
    
    def test_batch_empty(self):
        """Test empty batch returns 400."""
        response = client.post("/v1/romanize/batch", json={
            "texts": []
        })
        assert response.status_code == 400
    
    def test_batch_too_large(self):
        """Test batch over limit returns 400."""
        response = client.post("/v1/romanize/batch", json={
            "texts": ["test"] * 101
        })
        assert response.status_code == 400


class TestLanguagesEndpoint:
    """Tests for languages endpoint."""
    
    def test_list_languages(self):
        """Test listing supported languages."""
        response = client.get("/v1/languages")
        assert response.status_code == 200
        data = response.json()
        assert "languages" in data
        assert "styles" in data
        # Verify expected languages present
        lang_codes = [l["code"] for l in data["languages"]]
        assert "hi" in lang_codes  # Hindi
        assert "ko" in lang_codes  # Korean
```

```yaml
name: Test

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main, develop]

jobs:
  test:
    runs-on: ubuntu-latest
    
    strategy:
      matrix:
        python-version: ['3.11', '3.12']
    
    steps:
    - uses: actions/checkout@v4
    
    - name: Set up Python ${{ matrix.python-version }}
      uses: actions/setup-python@v5
      with:
        python-version: ${{ matrix.python-version }}
    
    - name: Install uv
      uses: astral-sh/setup-uv@v4
    
    - name: Install dependencies
      run: |
        uv sync --frozen --dev
    
    - name: Run tests
      run: |
        pytest tests/ -v --tb=short
    
    - name: Upload coverage
      uses: codecov/codecov-action@v4
      with:
        file: ./coverage.xml
```
