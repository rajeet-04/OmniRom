import re
from typing import Optional

class StyleTransformer:
    """Transform romanized text to different styles."""
    
    # Common diacritic mappings (standard -> academic)
    DIACRITIC_MAP = {
        'a': 'ā', 'A': 'Ā',
        'e': 'ē', 'E': 'Ē',
        'i': 'ī', 'I': 'Ī',
        'o': 'ō', 'O': 'Ō',
        'u': 'ū', 'U': 'Ū',
        # Common combinations
        'ai': 'ai',  # Keep as is in academic
        'au': 'au',
    }
    
    # Informal reductions (standard -> chat)
    CHAT_MAP = {
        'ā': 'a', 'ē': 'e', 'ī': 'i', 'ō': 'o', 'ū': 'u',
        'Ā': 'A', 'Ē': 'E', 'Ī': 'I', 'Ō': 'O', 'Ū': 'U',
        'sh': 'sh',  # Keep sh
        'th': 'th',
        # Common word endings
        'desu': 'des',
        'masu': 'mas',
        'nite': 'nai',
        'ndesu': 'ndes',
    }
    
    def transform(self, text: str, style: str) -> str:
        """Transform text to requested style."""
        if style == "standard":
            return text
        elif style == "academic":
            return self._to_academic(text)
        elif style == "chat":
            return self._to_chat(text)
        elif style == "phonetic":
            return self._to_phonetic(text)
        return text
    
    def _to_academic(self, text: str) -> str:
        """Add proper diacritics for academic style."""
        # For Indic languages, use proper transliteration
        # This is a simplified version - ideally would be engine-specific
        result = text
        
        # Add macrons to long vowels (common patterns)
        # This is a basic implementation
        patterns = [
            (r'\baa\b', 'ā'),  # aa -> ā (word boundary)
            (r'\bee\b', 'ē'),
            (r'\bioo\b', 'ī'),
            (r'\boo\b', 'ō'),
            (r'\buu\b', 'ū'),
        ]
        
        for pattern, replacement in patterns:
            result = re.sub(pattern, replacement, result, flags=re.IGNORECASE)
        
        return result
    
    def _to_chat(self, text: str) -> str:
        """Convert to informal chat style."""
        result = text
        
        # Remove diacritics for casual style
        replacements = {
            'ā': 'a', 'ē': 'e', 'ī': 'i', 'ō': 'o', 'ū': 'u',
            'Ā': 'A', 'Ē': 'E', 'Ī': 'I', 'Ō': 'O', 'Ū': 'U',
            'ṇ': 'n', 'ṅ': 'n', 'ñ': 'n',
            'ś': 's', 'ṣ': 's',
            'ṭ': 't', 'ḍ': 'd',
        }
        
        for old, new in replacements.items():
            result = result.replace(old, new)
        
        return result
    
    def _to_phonetic(self, text: str) -> str:
        """Simplified phonetic approximation."""
        # Use Unidecode for phonetic
        try:
            from unidecode import unidecode
            return unidecode(text)
        except ImportError:
            return self._to_chat(text)


_styler: Optional[StyleTransformer] = None

def get_styler() -> StyleTransformer:
    """Get style transformer singleton."""
    global _styler
    if _styler is None:
        _styler = StyleTransformer()
    return _styler

import requests

class LLM_styler:
    """Use LLM for intelligent style transformation."""
    
    def __init__(self, endpoint: str = None):
        # Use Ollama local endpoint
        self.endpoint = endpoint or os.getenv("OLLAMA_ENDPOINT", "http://localhost:11434")
    
    def transform(self, text: str, style: str) -> str:
        """Use LLM to transform text style."""
        prompt = f"""Transform this text to a more conversational, chat-friendly style.
Original: {text}
Style: {style}"""
        
        response = requests.post(
            f"{self.endpoint}/api/generate",
            json={"model": "llama3", "prompt": prompt, "stream": False}
        )
        return response.json().get("response", text)
