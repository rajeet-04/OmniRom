```python
from typing import Optional

class KoreanEngine:
    """Korean Hangul romanization."""
    
    def __init__(self):
        self.romanizer = None
        self.available = False
        self._load_romanizer()
    
    def _load_romanizer(self):
        """Load Hangul romanizer."""
        try:
            from hangul_romanizer import Romanizer
            self.romanizer = Romanizer()
            self.available = True
        except ImportError:
            self.available = False
    
    def romanize(self, text: str, style: str = "revisied") -> str:
        """
        Romanize Korean Hangul text.
        
        Args:
            text: Input Korean text
            style: 'revised' (official), 'mccune-reischauer' (older)
        
        Returns:
            Romanized text
        """
        if not text:
            return ""
        
        if not self.available:
            return self._fallback_romanize(text)
        
        try:
            if hasattr(self.romanizer, 'romanize'):
                result = self.romanizer.romanize(text)
                return result
            else:
                return self._fallback_romanize(text)
        except Exception:
            return self._fallback_romanize(text)
    
    def _fallback_romanize(self, text: str) -> str:
        """
        Fallback: Basic Unicode Hangul to Latin mapping.
        Uses algorithmic conversion for when library unavailable.
        """
        # Hangul Unicode ranges:
        # Leading consonants: 0x1100-0x115F
        # Vowels: 0x1160-0x11A7
        # Trailing consonants: 0x11A8-0x11FF
        # Precomposed Hangul: 0xAC00-0xD7AF
        
        # For simplicity, use Unidecode as fallback
        try:
            from unidecode import unidecode
            return unidecode(text)
        except ImportError:
            # Last resort: return as-is
            return text


_korean_engine: Optional[KoreanEngine] = None

def get_korean_engine() -> KoreanEngine:
    """Get or create Korean engine singleton."""
    global _korean_engine
    if _korean_engine is None:
        _korean_engine = KoreanEngine()
    return _korean_engine
```
