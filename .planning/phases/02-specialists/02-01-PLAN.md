from typing import Optional

class IndicEngine:
    """Indic language transliteration using AI4Bharat IndicXlit."""
    
    # Map script types to language codes
    SCRIPT_LANG_MAP = {
        "deva": "hi",  # Devanagari -> Hindi
        "beng": "bn",  # Bengali
        "taml": "ta",  # Tamil
        "telu": "te",  # Telugu
        "mlym": "ml",  # Malayalam
        "knda": "kn",  # Kannada
        "gujr": "gu",  # Gujarati
        "orya": "or",  # Oriya
        "guru": "pa",  # Gurmukhi/Punjabi
        "asom": "as",  # Assamese
        "mtei": "mni", # Meitei (Manipuri)
    }
    
    def __init__(self):
        self.model = None
        self.available = False
        self._load_model()
    
    def _load_model(self):
        """Load AI4Bharat model."""
        try:
            # Try to import ai4bharat transliteration
            from ai4bharat.transliteration import Script
            self.Script = Script
            self.available = True
        except ImportError:
            # Fallback: will use simpler transliteration
            self.available = False
    
    def is_supported_script(self, script_code: str) -> bool:
        """Check if script is supported."""
        return script_code.lower() in self.SCRIPT_LANG_MAP
    
    def get_lang_code(self, script_code: str) -> Optional[str]:
        """Get language code for script."""
        return self.SCRIPT_LANG_MAP.get(script_code.lower())
    
    def romanize(self, text: str, script_code: str) -> str:
        """
        Romanize Indic text.
        
        Args:
            text: Input text in Indic script
            script_code: ISO 15924 script code (deva, beng, taml, etc.)
        
        Returns:
            Romanized text
        """
        if not text or not self.available:
            return text
        
        lang_code = self.get_lang_code(script_code)
        if not lang_code:
            return text
        
        try:
            # Use AI4Bharat
            result = self.Script.get_instance(lang_code)
            romanized = result.transliterate(text, max_result_len=len(text) * 3)
            return romanized
        except Exception:
            return text


_indic_engine: Optional[IndicEngine] = None

def get_indic_engine() -> IndicEngine:
    """Get or create Indic engine singleton."""
    global _indic_engine
    if _indic_engine is None:
        _indic_engine = IndicEngine()
    return _indic_engine

from typing import Optional

class IndicFallback:
    """Simple fallback for Indic scripts when AI4Bharat unavailable."""
    
    # Basic Devanagari to Latin mapping
    DEVA_MAP = {
        'अ': 'a', 'आ': 'aa', 'इ': 'i', 'ई': 'ee', 'उ': 'u', 'ऊ': 'oo',
        'ए': 'e', 'ऐ': 'ai', 'ओ': 'o', 'औ': 'au',
        'क': 'k', 'ख': 'kh', 'ग': 'g', 'घ': 'gh', 'ङ': 'ng',
        'च': 'c', 'छ': 'ch', 'ज': 'j', 'झ': 'jh', 'ञ': 'ny',
        'ट': 't', 'ठ': 'th', 'ड': 'd', 'ढ': 'dh', 'ण': 'n',
        'त': 't', 'थ': 'th', 'द': 'd', 'ध': 'dh', 'न': 'n',
        'प': 'p', 'फ': 'ph', 'ब': 'b', 'भ': 'bh', 'म': 'm',
        'य': 'y', 'र': 'r', 'ल': 'l', 'व': 'v', 'श': 'sh',
        'ष': 'sh', 'स': 's', 'ह': 'h',
        'ा': 'aa', 'ि': 'i', 'ी': 'ee', 'ु': 'u', 'ू': 'oo',
        'े': 'e', 'ै': 'ai', 'ो': 'o', 'ौ': 'au',
        '्': '', 'ं': 'n', 'ः': 'h',
    }
    
    @staticmethod
    def romanize(text: str, script: str = "deva") -> str:
        """Simple character-by-character romanization."""
        if not text:
            return ""
        
        mapping = IndicFallback.DEVA_MAP if script == "deva" else {}
        
        result = []
        for char in text:
            result.append(mapping.get(char, char))
        
        return ''.join(result)
```

from app.engines.uroman_engine import get_uroman_engine
from app.engines.indic_engine import get_indic_engine, IndicEngine
from app.engines.indic_fallback import IndicFallback

# Scripts that Indic engine handles better
INDIC_SCRIPTS = {
    "deva", "beng", "taml", "telu", "mlym", "knda", 
    "gujr", "orya", "guru", "asom", "mtei"
}

def route_romanization(text: str, script_type: str, script_code: str = None) -> Tuple[str, str]:
    """
    Route text to the best romanization engine.
    
    Returns:
        Tuple of (romanized_text, engine_name)
    """
    # Priority 1: Indic scripts -> use Indic engine
    if script_code and script_code.lower() in INDIC_SCRIPTS:
        indic = get_indic_engine()
        if indic.available:
            try:
                romanized = indic.romanize(text, script_code)
                if romanized and romanized != text:
                    return romanized, "indic-xlit"
            except Exception:
                pass
        # Fallback to simple Indic fallback
        try:
            romanized = IndicFallback.romanize(text, script_code)
            if romanized and romanized != text:
                return romanized, "indic-fallback"
        except Exception:
            pass
    
    # Priority 2: ICU for supported scripts
    icu_engine = get_icu_engine()
    if script_type in SCRIPT_ENGINE_MAP and icu_engine.is_supported(script_type):
        romanized = icu_engine.romanize(text, script_type)
        return romanized, "icu"
    
    # Priority 3: uroman fallback
    uroman = get_uroman_engine()
    romanized = uroman.romanize(text)
    return romanized, "uroman"
```
