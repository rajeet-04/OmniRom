from typing import Optional
from pypinyin import pinyin, Style, load_phrases_dict, load_single_dict

class ChineseEngine:
    """Chinese to Pinyin romanization."""
    
    def __init__(self):
        self.style = Style.TONE3  # Default: tone numbers
    
    def romanize(self, text: str, style: str = "standard") -> str:
        """
        Romanize Chinese text to Pinyin.
        
        Args:
            text: Input Chinese text
            style: 'standard' (tone marks), 'numbers' (tone numbers), 'first_letter'
        
        Returns:
            Pinyin romanization
        """
        if not text:
            return ""
        
        # Select style
        if style == "numbers":
            pinyin_style = Style.TONE3
        elif style == "first_letter":
            pinyin_style = Style.FIRST_LETTER
        else:  # standard / academic
            pinyin_style = Style.TONE_MARK
            # Also map to tone marks if possible
            self.style = Style.TONE_MARK
        
        try:
            result = pinyin(text, style=pinyin_style)
            pinyin_str = ' '.join([item[0] for item in result])
            return pinyin_str
        except Exception:
            return text
    
    def get_pinyin(self, text: str) -> str:
        """Get pinyin with tone numbers."""
        result = pinyin(text, style=Style.TONE3)
        return ' '.join([item[0] for item in result])
    
    def get_tone_mark(self, text: str) -> str:
        """Get pinyin with tone marks."""
        result = pinyin(text, style=Style.TONE_MARK)
        return ' '.join([item[0] for item in result])


_chinese_engine: Optional[ChineseEngine] = None

def get_chinese_engine() -> ChineseEngine:
    """Get or create Chinese engine singleton."""
    global _chinese_engine
    if _chinese_engine is None:
        _chinese_engine = ChineseEngine()
    return _chinese_engine

from typing import Optional

class JapaneseEngine:
    """Japanese romanization using Cutlet (MeCab-based)."""
    
    def __init__(self):
        self.cutlet = None
        self.available = False
        self._load_engine()
    
    def _load_engine(self):
        """Load Cutlet engine."""
        try:
            import cutlet
            self.cutlet = cutlet.Cutlet()
            self.cutlet.meta_infos = False
            self.available = True
        except ImportError:
            # Fallback will be created
            self.available = False
    
    def romanize(self, text: str, style: str = "hepburn") -> str:
        """
        Romanize Japanese text.
        
        Args:
            text: Input Japanese text (kanji, kana, or mixed)
            style: 'hepburn', 'kunrei', or 'nihon'
        
        Returns:
            Romanized text
        """
        if not text:
            return ""
        
        if not self.available:
            return self._fallback_romanize(text)
        
        try:
            # Select romanization system
            if style == "hepburn":
                self.cutlet.romaji_system = "hepburn"
            elif style == "kunrei":
                self.cutlet.romaji_system = "kunrei"
            elif style == "nihon":
                self.cutlet.romaji_system = "nihon"
            
            result = self.cutlet.romaji(text)
            return result
        except Exception:
            return self._fallback_romanize(text)
    
    def _fallback_romaji(self, text: str) -> str:
        """Simple fallback using character mapping."""
        # Basic hiragana/katakana to romaji mapping
        hira_map = {
            'あ': 'a', 'い': 'i', 'う': 'u', 'え': 'e', 'お': 'o',
