```python
import subprocess
import os
from typing import Optional

class UromanEngine:
    """Universal romanizer using uroman."""
    
    def __init__(self):
        self.available = self._check_availability()
    
    def _check_availability(self) -> bool:
        """Check if uroman is available."""
        try:
            # Try to run uroman --help
            result = subprocess.run(
                ["uroman", "--help"],
                capture_output=True,
                timeout=5
            )
            return result.returncode == 0
        except (FileNotFoundError, subprocess.TimeoutExpired):
            return False
    
    def romanize(self, text: str) -> str:
        """
        Romanize text using uroman.
        
        Falls back to basic character stripping if uroman unavailable.
        """
        if not text:
            return ""
        
        if not self.available:
            # Fallback: basic ASCII conversion
            return self._basic_romanize(text)
        
        try:
            result = subprocess.run(
                ["uroman"],
                input=text,
                capture_output=True,
                text=True,
                timeout=30
            )
            if result.returncode == 0:
                return result.stdout.strip()
            else:
                return self._basic_romanize(text)
        except (subprocess.TimeoutExpired, Exception):
            return self._basic_romanize(text)
    
    def _basic_romanize(self, text: str) -> str:
        """Basic fallback: keep only Latin characters."""
        import unicodedata
        # Normalize and keep letters
        normalized = unicodedata.normalize('NFD', text)
        latin = ''.join(c for c in normalized if unicodedata.category(c) != 'Mn')
        return latin


_uroman_engine: Optional[UromanEngine] = None

def get_uroman_engine() -> UromanEngine:
    """Get or create uroman engine singleton."""
    global _uroman_engine
    if _uroman_engine is None:
        _uroman_engine = UromanEngine()
    return _uroman_engine
```

```python
from typing import Tuple, Optional
from app.engines.icu_engine import get_icu_engine
from app.engines.uroman_engine import get_uroman_engine

# Map script types to engines
SCRIPT_ENGINE_MAP = {
    "cyrillic": "icu",
    "greek": "icu",
    "arabic": "icu",
    "hebrew": "icu",
    # Add more as supported
}

def route_romanization(text: str, script_type: str) -> Tuple[str, str]:
    """
    Route text to the best romanization engine.
    
    Returns:
        Tuple of (romanized_text, engine_name)
    """
    # Check if ICU supports this script
    icu_engine = get_icu_engine()
    if script_type in SCRIPT_ENGINE_MAP and icu_engine.is_supported(script_type):
        romanized = icu_engine.romanize(text, script_type)
        return romanized, "icu"
    
    # Fall back to uroman for unsupported scripts
    uroman = get_uroman_engine()
    romanized = uroman.romanize(text)
    return romanized, "uroman"


def get_supported_engines() -> dict:
    """Get list of supported engines and their capabilities."""
    icu = get_icu_engine()
    return {
        "icu": {
            "supported_scripts": list(icu.transliterators.keys()),
            "display_name": "ICU (International Components for Unicode)"
        },
        "uroman": {
            "supported_scripts": ["universal"],
            "display_name": "uroman (Universal Romanizer)"
        }
    }
```

```python
from fastapi import APIRouter, HTTPException
import time
from app.schemas.romanize import RomanizeRequest, RomanizeResponse
from app.core.detector import detect_script, get_script_type
from app.engines.router import route_romanization, get_supported_engines
from app.engines.icu_engine import get_icu_engine

router = APIRouter(prefix="/v1", tags=["romanize"])

@router.post("/romanize", response_model=RomanizeResponse)
async def romanize(request: RomanizeRequest):
    """Romanize text from any script to Latin."""
    if not request.text or not request.text.strip():
        raise HTTPException(status_code=400, detail="Text cannot be empty")
    
    start_time = time.time()
    
    try:
        # Detect script
        detection = detect_script(request.text)
        script_type = get_script_type(detection["language"], detection["script"])
        
        # Route to best engine
        romanized, engine_used = route_romanization(request.text, script_type)
        
        processing_time_ms = int((time.time() - start_time) * 1000)
        
        return RomanizeResponse(
            original=request.text,
            romanized=romanized,
            metadata={
                "detected_lang": detection["language"],
                "detected_script": detection["script"],
                "engine_used": engine_used,
                "cached": False,
                "processing_time_ms": processing_time_ms
            }
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Romanization failed: {str(e)}")

@router.get("/engines")
async def list_engines():
    """List available romanization engines."""
    return get_supported_engines()
```
