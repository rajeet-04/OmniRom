```python
import pycld3
from typing import Optional

def detect_script(text: str) -> dict:
    """
    Detect the script/language of input text.
    
    Returns dict with:
    - language: ISO 639-1 code
    - script: Script name
    - reliable: Whether detection is reliable
    """
    if not text or not text.strip():
        return {"language": "und", "script": "Unknown", "reliable": False}
    
    # Use pycld3 for detection
    detector = pycld3.LanguageIdentifier()
    result = detector.FindLanguage(text)
    
    return {
        "language": result.language,
        "script": result.script,
        "reliable": result.is_reliable,
        "probability": result.probability
    }

def get_script_type(language: str, script: str) -> str:
    """
    Categorize the script type for routing.
    Returns: 'cyrillic', 'greek', 'arabic', 'cjk', 'hangul', 'hiragana', 'katakana', 'latin', 'other'
    """
    script_lower = script.lower() if script else ""
    
    if script_lower in ("cyrillic",):
        return "cyrillic"
    elif script_lower in ("greek",):
        return "greek"
    elif script_lower in ("arabic", "hebrew"):
        return "arabic"
    elif script_lower in ("hans", "hant", "hanzi", "kanji"):
        return "cjk"
    elif script_lower == "hangul":
        return "hangul"
    elif script_lower == "hiragana":
        return "hiragana"
    elif script_lower == "katakana":
        return "katakana"
    elif script_lower in ("latn", "latin"):
        return "latin"
    else:
        return "other"
```

```python
from PyICU import Transliterator
from typing import Optional

class ICUEngine:
    """Rule-based transliteration using ICU library."""
    
    def __init__(self):
        # Load transliterator for various scripts
        self.transliterators = {
            "cyrillic": Transliterator.createInstance("Cyrillic-Latin"),
            "greek": Transliterator.createInstance("Greek-Latin"),
            "arabic": Transliterator.createInstance("Arabic-Latin"),
            "hebrew": Transliterator.createInstance("Hebrew-Latin"),
        }
        # Default fallback transliterator
        self.default = Transliterator.createInstance("Any-Latin")
    
    def romanize(self, text: str, script_type: str) -> str:
        """
        Romanize text using appropriate ICU transliterator.
        
        Args:
            text: Input text to romanize
            script_type: Detected script type (cyrillic, greek, arabic, etc.)
        
        Returns:
            Romanized text
        """
        if not text:
            return ""
        
        transliterator = self.transliterators.get(script_type, self.default)
        return transliterator.transliterate(text)
    
    def is_supported(self, script_type: str) -> bool:
        """Check if script is supported by ICU."""
        return script_type in self.transliterators


# Singleton instance
_icu_engine: Optional[ICUEngine] = None

def get_icu_engine() -> ICUEngine:
    """Get or create ICU engine singleton."""
    global _icu_engine
    if _icu_engine is None:
        _icu_engine = ICUEngine()
    return _icu_engine
```

```python
class RomanizeRequest(BaseModel):

    source_lang: Optional[str] = Field("auto", description="Source language (auto-detect if not specified)")
```

```python
from fastapi import APIRouter, HTTPException
from app.schemas.romanize import RomanizeRequest, RomanizeResponse
from app.core.detector import detect_script, get_script_type
from app.engines.icu_engine import get_icu_engine

router = APIRouter(prefix="/v1", tags=["romanize"])

@router.post("/romanize", response_model=RomanizeResponse)
async def romanize(request: RomanizeRequest):
    """Romanize text from any script to Latin."""
    if not request.text or not request.text.strip():
        raise HTTPException(status_code=400, detail="Text cannot be empty")
    
    # Detect script
    detection = detect_script(request.text)
    script_type = get_script_type(detection["language"], detection["script"])
    
    # Get engine and romanize
    engine = get_icu_engine()
    romanized = engine.romanize(request.text, script_type)
    
    return RomanizeResponse(
        original=request.text,
        romanized=romanized,
        metadata={
            "detected_lang": detection["language"],
            "detected_script": detection["script"],
            "engine_used": "icu",
            "cached": False
        }
    )
```
